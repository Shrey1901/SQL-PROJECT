CREATE TABLE DOG
(
DOG_ID NUMBER(5) PRIMARY KEY,
NAME VARCHAR(10) ,
BREED VARCHAR(20) NOT NULL,
AGE NUMBER(2) NOT NULL,
OWNER_ID VARCHAR(10),
SAFETY_BELT_ID VARCHAR(10),
ORGANIZATION_ID NUMBER(5),
CONSTRAINT OWNER_ID_FK FOREIGN KEY(OWNER_ID) REFERENCES OWNER(OWNER_ID),
CONSTRAINT SAFETY_BELT_ID_FK FOREIGN KEY(SAFETY_BELT_ID) REFERENCES SAFETY_BELT(SAFETY_BELT),
CONSTRAINT ORGANIZATION_ID FOREIGN KEY(ORGANIZATION_ID) REFERENCES ORGANIZATION(ORGANIZATION_ID)


)
